export async function getAllFiles() {
  const filesList = [];
  const dir = path.join(root, `data`);
  readFileList(dir,filesList);

  const result = []
  for (let i = 0; i< filesList.length; i++) {
    if (filesList[i].endsWith('.mdx')) {
      result.push(await getFileBySlug(filesList[i]));
    }
  }

  return result;

}
